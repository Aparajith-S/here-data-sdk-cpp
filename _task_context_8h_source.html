171  class TaskContextImpl : public Impl {
172  public:
177  using Response = client::ApiResponse<T, client::ApiError>;

196  client::CancellationContext context)
197  : execute_func_(std::move(execute_func)),
198  callback_(std::move(callback)),
199  context_(std::move(context)),
200  state_{State::PENDING} {}

208  void Execute() override {
209  State expected_state = State::PENDING;
210 
211  if (!state_.compare_exchange_strong(expected_state, State::IN_PROGRESS)) {
212  return;
213  }
214 
215  // Moving the user callback and function guarantee that they are
216  // executed exactly once
217  ExecuteFunc function = nullptr;
218  UserCallback callback = nullptr;
219 
220  {
221  std::lock_guard<std::mutex> lock(mutex_);
222  function = std::move(execute_func_);
223  callback = std::move(callback_);
224  }
225 
226  Response user_response =
227  client::ApiError(client::ErrorCode::Cancelled, "Cancelled");
228 
229  if (function && !context_.IsCancelled()) {
230  auto response = function(context_);
231  // Cancel could occur during the function execution. In that case,
232  // ignore the response.
233  if (!context_.IsCancelled() ||
234  (!response.IsSuccessful() &&
235  response.GetError().GetErrorCode() == ErrorCode::RequestTimeout)) {
236  user_response = std::move(response);
237  }
238  }
239 
240  if (callback) {
241  callback(std::move(user_response));
242  }
243 
244  // Resources need to be released before the notification, else lambas
245  // would have captured resources like network or `TaskScheduler`.
246  function = nullptr;
247  callback = nullptr;
248 
249  condition_.Notify();
250 
251  state_.store(State::COMPLETED);
252  }
253 
262  bool BlockingCancel(std::chrono::milliseconds timeout) override {
263  if (state_.load() == State::COMPLETED) {
264  return true;
265  }
266 
267  // Cancels the operation and waits for the notification.
268  if (!context_.IsCancelled()) {
269  context_.CancelOperation();
270  }
271 
272  {
273  std::lock_guard<std::mutex> lock(mutex_);
274  execute_func_ = nullptr;
275  }
276 
277  return condition_.Wait(timeout);
278  }
279 
285  client::CancellationToken CancelToken() override {
286  auto context = context_;
287  return client::CancellationToken(
288  [context]() mutable { context.CancelOperation(); });
289  }

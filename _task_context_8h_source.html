112  template <typename T>
113  class TaskContextImpl : public Impl {
114  public:
115  using Response = client::ApiResponse<T, client::ApiError>;
116  using ExecuteFunc = std::function<Response(client::CancellationContext)>;
117  using UserCallback = std::function<void(Response)>;
118 
119  TaskContextImpl(ExecuteFunc execute_func, UserCallback callback,
120  client::CancellationContext context)
121  : execute_func_(std::move(execute_func)),
122  callback_(std::move(callback)),
123  context_(std::move(context)),
124  state_{State::PENDING} {}
125 
126  ~TaskContextImpl() {}
127 
128  void Execute() override {
129  State expected_state = State::PENDING;
130 
131  if (!state_.compare_exchange_strong(expected_state, State::IN_PROGRESS)) {
132  return;
133  }
134 
135  // Moving the user callback and function guarantee that they will be
136  // executed exactly once
137  ExecuteFunc function = nullptr;
138  UserCallback callback = nullptr;
139 
140  {
141  std::lock_guard<std::mutex> lock(mutex_);
142  function = std::move(execute_func_);
143  callback = std::move(callback_);
144  }
145 
146  Response user_response =
147  client::ApiError(client::ErrorCode::Cancelled, "Cancelled");
148 
149  if (function && !context_.IsCancelled()) {
150  auto response = function(context_);
151  // Cancel could occur during function execution, in that case we ignore
152  // the response.
153  if (!context_.IsCancelled() ||
154  (!response.IsSuccessful() &&
155  response.GetError().GetErrorCode() == ErrorCode::RequestTimeout)) {
156  user_response = std::move(response);
157  }
158  }
159 
160  if (callback) {
161  callback(std::move(user_response));
162  }
163 
164  // Resources needs to be released before notification, else lambas would
165  // have captured resources like network or TaskScheduler
166  function = nullptr;
167  callback = nullptr;
168 
169  condition_.Notify();
170 
171  state_.store(State::COMPLETED);
172  }
173 
174  bool BlockingCancel(std::chrono::milliseconds timeout) override {
175  if (state_.load() == State::COMPLETED) {
176  return true;
177  }
178 
179  // Cancel operation and wait for notification
180  if (!context_.IsCancelled()) {
181  context_.CancelOperation();
182  }
183 
184  {
185  std::lock_guard<std::mutex> lock(mutex_);
186  execute_func_ = nullptr;
187  }
188 
189  return condition_.Wait(timeout);
190  }
191 
192  client::CancellationToken CancelToken() override {
193  auto context = context_;
194  return client::CancellationToken(
195  [context]() mutable { context.CancelOperation(); });
196  }
197 
198  enum class State { PENDING, IN_PROGRESS, COMPLETED };

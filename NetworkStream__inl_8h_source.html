82 inline bool IsContentLengthHeader(const std::string& key) {
83  static constexpr char content_length[] = "content-length";
84  if (sizeof(content_length) - 1 != key.size()) return false;
85  const char* p = content_length;
86  for (const char& c : key) {
87  if (*p++ != (0x20 | c)) return false;
88  }
89  return true;
90 }

92 inline bool StrToSize_t(const std::string& value, std::size_t& result) {
93  const unsigned char* ptr =
94  reinterpret_cast<const unsigned char*>(value.data());
95  const unsigned char* end = ptr + value.size();
96  constexpr std::size_t base = 10;
97  std::size_t result_ = 0;
98  for (; end > ptr; ++ptr) {
99  std::size_t tmp = *ptr - '0';
100  if (9 < tmp) return false;
101  tmp += result_ * base;
102  if (tmp < result_) return false;
103  result_ = tmp;
104  }
105  result = result_;
106  return true;
107 }
